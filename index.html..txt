/* ════════════════════════════════════════
   ④ Google Sheets API
   
   [사용 방법]
   - SHEET_ID  : 스프레드시트 URL의 /d/XXXX/edit 에서 XXXX 부분
   - API_KEY   : Google Cloud Console에서 발급한 API 키
   
   점수 저장 시트 구조 (시트명: scores):
   A열: 이름 | B열: 점수 | C열: 총문제 | D열: 정답률 | E열: 시간
════════════════════════════════════════ */
let SHEET_ID = ld('smb_sheet_id', '');
let API_KEY  = ld('smb_api_key', '');

// ── 점수 저장 (Sheets API append) ──
async function saveScoreToSheet(name, corr, total, pct) {
  if (!SHEET_ID || !API_KEY) {
    // 설정 없으면 localStorage 저장만
    saveScoreLocal(name, corr, total, pct);
    return { ok: false, msg: 'Sheets 미설정 — 로컬 저장' };
  }
  const now = new Date();
  const ts = `${now.getFullYear()}-${String(now.getMonth()+1).padStart(2,'0')}-${String(now.getDate()).padStart(2,'0')} ${String(now.getHours()).padStart(2,'0')}:${String(now.getMinutes()).padStart(2,'0')}`;
  const url = `https://sheets.googleapis.com/v4/spreadsheets/${SHEET_ID}/values/scores!A:E:append?valueInputOption=USER_ENTERED&key=${API_KEY}`;
  try {
    const res = await fetch(url, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ values: [[name, corr, total, pct, ts]] })
    });
    if (!res.ok) throw new Error(await res.text());
    saveScoreLocal(name, corr, total, pct); // 로컬 캐시도 저장
    return { ok: true };
  } catch(e) {
    console.error('Sheets 저장 실패:', e);
    saveScoreLocal(name, corr, total, pct);
    return { ok: false, msg: e.message };
  }
}

// ── 점수 불러오기 (Sheets API get) ──
async function loadScoresFromSheet() {
  if (!SHEET_ID || !API_KEY) return null;
  const url = `https://sheets.googleapis.com/v4/spreadsheets/${SHEET_ID}/values/scores!A2:E?key=${API_KEY}`;
  try {
    const res = await fetch(url);
    if (!res.ok) throw new Error(await res.text());
    const data = await res.json();
    const rows = (data.values || []).map(r => ({
      name: r[0]||'', corr: Number(r[1])||0,
      total: Number(r[2])||8, pct: Number(r[3])||0, ts: r[4]||''
    }));
    rows.sort((a,b) => b.corr !== a.corr ? b.corr-a.corr : b.pct-a.pct);
    sv('smb_hall', rows); // 로컬 캐시 갱신
    return rows;
  } catch(e) {
    console.error('Sheets 불러오기 실패:', e);
    return null;
  }
}

// ── 로컬 점수 저장 ──
function saveScoreLocal(name, corr, total, pct) {
  const now = new Date();
  const ts = `${now.getMonth()+1}/${now.getDate()} ${String(now.getHours()).padStart(2,'0')}:${String(now.getMinutes()).padStart(2,'0')}`;
  let hall = ld('smb_hall', []);
  hall.push({name, corr, total, pct, ts});
  hall.sort((a,b) => b.corr !== a.corr ? b.corr-a.corr : b.pct-a.pct);
  if (hall.length > 50) hall = hall.slice(0, 50);
  sv('smb_hall', hall);
}

function shuffle(a) {
  const r=[...a];
  for(let i=r.length-1;i>0;i--){const j=Math.floor(Math.random()*(i+1));[r[i],r[j]]=[r[j],r[i]];}
  return r;
}
function makeChoices(c) {
  return shuffle([c, ...shuffle(questions.map(q=>q.name).filter(n=>n!==c)).slice(0,3)]);
}

function pick(idx) {
  if (answered) return;
  const q    = quizQs[current];
  const isOk = q.choices[idx] === q.name;

  if (!isOk && !q.retried) {
    // 첫 번째 오답 → 키워드 힌트 + 다시 풀어볼래?
    const b = document.getElementById('cb'+idx);
    b.classList.add('wrong');
    b.disabled = true;
    const fb = document.getElementById('fb');
    fb.classList.remove('hidden');
    fb.classList.add('ng');
    fb.innerHTML = `❌ 아쉬워요! 힌트를 보고 다시 도전해봐요!<br><br>
      <span style="font-size:13px;font-weight:800;color:#0d5c3a;background:#e4f5ec;padding:6px 12px;border-radius:8px;display:inline-block;margin-top:4px">
        ${q.keyword||''}
      </span>`;
    document.getElementById('nav').innerHTML =
      `<button class="btn pri" onclick="retry(${idx})">다시 풀어볼래? 🔄</button>`;
    return;
  }

  // 정답이거나 두 번째 시도
  answered = true;
  // 두 번째 시도(retried)면 첫 번째에 이미 틀렸으므로 오답 처리
  scores[current] = q.retried ? false : isOk;
  q.choices.forEach((c,i) => {
    const b = document.getElementById('cb'+i);
    b.disabled = true;
    if (c === q.name) b.classList.add(isOk && i===idx ? 'correct' : 'reveal');
    if (i === idx && !isOk) b.classList.add('wrong');
    const vb = document.getElementById('verb'+i);
    if (vb) vb.style.display = 'inline';
  });
  const fb = document.getElementById('fb');
  fb.classList.remove('hidden', 'ok', 'ng');
  fb.classList.add(isOk ? 'ok' : 'ng');
  fb.innerHTML = (isOk ? '✅ ' : '❌ ') + (isOk ? q.ok : q.ng);
  const isLast = current === quizQs.length - 1;
  document.getElementById('nav').innerHTML =
    `<button class="btn pri" onclick="nextQ()">${isLast ? '결과 보기' : '다음 문제'} →</button>`;

function retry(wrongIdx) {
  const q = quizQs[current];
  q.retried = true; // 두 번째 기회 표시
  // 틀린 버튼 비활성 유지, 나머지 다시 활성화
  q.choices.forEach((c,i) => {
    if (i !== wrongIdx) {
      const b = document.getElementById('cb'+i);
      b.disabled = false;
      b.classList.remove('wrong','correct','reveal');
    }
  });
  const fb = document.getElementById('fb');
  fb.classList.add('hidden');
  document.getElementById('nav').innerHTML = '';
}

    return `<div class="ritem ${rc}">
      <div class="rnum ${nc}">${medal}</div>
      <div class="rinfo">
        <div class="rname">${r.name} ${bdg}</div>
        <div class="rmeta">${r.ts}</div>
      </div>
      <div style="text-align:right">
        <div class="rsn">${r.corr}<span style="font-size:13px;color:var(--text3)">/${r.total}</span></div>
        <div class="rsp">${r.pct}%</div>
      </div>
    </div>`;
  }).join('');
  el.innerHTML = `
    <div class="hall-hdr">
      <div class="hall-ttl">🏆 명예의 전당</div>
      <button class="clr-btn" onclick="clearHall()">로컬 초기화</button>
    </div>
    <div class="rank-list">${rows}</div>`;
}

async function testSheets() {
  const msg = document.getElementById('sheets-test-msg');
  const sid = document.getElementById('cfg-sheet-id').value.trim();
  const key = document.getElementById('cfg-api-key').value.trim();
  if (!sid || !key) { msg.style.color='var(--red)'; msg.textContent='ID와 API Key를 먼저 입력하세요.'; return; }
  msg.style.color='var(--text2)'; msg.textContent='연결 테스트 중...';
  try {
    const res = await fetch(`https://sheets.googleapis.com/v4/spreadsheets/${sid}?key=${key}`);
    if (res.ok) { msg.style.color='var(--green)'; msg.textContent='✅ 연결 성공!'; }
    else        { const e=await res.json(); msg.style.color='var(--red)'; msg.textContent='❌ '+e.error?.message; }
  } catch(e) {
    msg.style.color='var(--red)'; msg.textContent='❌ 네트워크 오류';
  }
}
function renderQList() {
  document.getElementById('q-count').textContent = questions.length;
  const el = document.getElementById('q-list');
  if (!questions.length) {
    el.innerHTML = '<div style="font-size:13px;color:var(--text3);text-align:center;padding:14px">등록된 문제가 없어요.</div>';
    return;
  }
  el.innerHTML = questions.map((q,i) => `
    <div class="q-item">
      <div>
        <div class="q-item-name">${q.name}</div>
        <div class="q-item-verb">${q.verb}</div>
      </div>
      <div class="q-item-acts">
        <button class="ibtn" onclick="openEdit(${i})">수정</button>
        <button class="ibtn del" onclick="delQ(${i})">삭제</button>
      </div>
    </div>`).join('');
}

function openEdit(idx) {
  const q = questions[idx];
  document.getElementById('form-title').textContent = '문제 수정';
  document.getElementById('edit-idx').value  = idx;
  document.getElementById('f-name').value    = q.name;
  document.getElementById('f-verb').value    = q.verb;
  document.getElementById('f-scene').value   = q.scene;
  document.getElementById('f-hint').value    = q.hint||'';
  document.getElementById('f-ok').value      = q.ok;
  document.getElementById('f-ng').value      = q.ng;
  document.getElementById('edit-form').style.display = '';
  document.getElementById('edit-form').scrollIntoView({behavior:'smooth'});
}
function saveQ() {
  const name  = document.getElementById('f-name').value.trim();
  const verb  = document.getElementById('f-verb').value.trim();
  const scene = document.getElementById('f-scene').value.trim();
  const hint  = document.getElementById('f-hint').value.trim();
  const ok    = document.getElementById('f-ok').value.trim();
  const ng    = document.getElementById('f-ng').value.trim();
  if (!name||!verb||!scene||!ok||!ng) { alert('필수 항목(*)을 모두 입력해주세요.'); return; }
  const q   = {name,verb,scene,hint,ok,ng};
  const idx = parseInt(document.getElementById('edit-idx').value);
  if (idx===-1) questions.push(q); else questions[idx]=q;
  sv('smb_q', questions);
  cancelEdit(); renderQList();
}
function delQ(idx) {
  if (!confirm(`"${questions[idx].name}" 문제를 삭제할까요?`)) return;
  questions.splice(idx,1); sv('smb_q', questions); renderQList();
}
